CREATE TABLE USER(
UID INT(100) UNSIGNED NOT NULL AUTO_INCREMENT,
ID VARCHAR(100) NOT NULL,
PW VARCHAR(100) NOT NULL,
NICKNAME VARCHAR(10) NOT NULL,
EQUIPMENT_WP_NO INT(4) UNSIGNED DEFAULT 0 NOT NULL,
EQUIPMENT_AR_NO INT(4) UNSIGNED DEFAULT 0 NOT NULL,
LV INT(6) UNSIGNED DEFAULT 1 NOT NULL,
STR_STATE INT(6) UNSIGNED DEFAULT 1 NOT NULL,
HP_STATE INT(6) UNSIGNED DEFAULT 1 NOT NULL,
DEF_STATE INT(6) UNSIGNED DEFAULT 1 NOT NULL,
MOVE INT(3) UNSIGNED DEFAULT 30 NOT NULL,
PRIMARY KEY(UID)
);

CREATE TABLE EQUIP_WP(
EQUIP_NO INT(3) UNSIGNED NOT NULL AUTO_INCREMENT,
ATK INT(6) UNSIGNED DEFAULT 1 NOT NULL,
EQUIP_NAME VARCHAR(100) NOT NULL,
RATE DOUBLE(7,4) UNSIGNED NOT NULL,
PRIMARY KEY(EQUIP_NO)
);

CREATE TABLE EQUIP_AM(
EQUIP_NO INT(3) UNSIGNED NOT NULL AUTO_INCREMENT,
DEF INT(6) UNSIGNED DEFAULT 1 NOT NULL,
HP INT(6) UNSIGNED DEFAULT 1 NOT NULL,
EQUIP_NAME VARCHAR(100) NOT NULL,
RATE DOUBLE(7,4) UNSIGNED NOT NULL,
PRIMARY KEY(EQUIP_NO)
);

CREATE TABLE Enemy(
EID INT(3) UNSIGNED NOT NULL AUTO_INCREMENT,
NAME VARCHAR(100) NOT NULL,
ATK INT(10) UNSIGNED DEFAULT 1 NOT NULL,
HP INT(15) UNSIGNED DEFAULT 1 NOT NULL,
DEF INT(7) UNSIGNED DEFAULT 1 NOT NULL,
ITEM_NO INT(3) NOT NULL,
RID INT(2) NOT NULL,
PRIMARY KEY(EID)
);
